    $.valuation = {};

    $.valuation.getValuation = function () {
        let url = 'https://biluppgifter.se/api/v1/valuation/valuate';
        let source = $('#vehicle-valuation-widget-template').html();
        let template;
        let valuationSection = $('#valuation-section-div');
        let loanWidget = $('#loan-widget');
        let loanBox = $('#box-loan');
        let valuationDiv = $('#valuation-div');

        let loanSource = $('#loan-template').html();
        let loanTemplate;

        axios.post(url, {
                id: 'kOaR6Mx83z'
            })
            .then(function (response) {
                if (response.data.length === 0) {
                    valuationDiv.fadeOut();
                    loanBox.fadeOut();

                    return;
                }

                template = Handlebars.compile(source);
                valuationSection.html(template(response.data));

                if (response.data.loan.payment_plan) {
                    let samblaData = {
                        loanSum: new Intl.NumberFormat('sv-SE', {
                            options: {
                                style: 'currency',
                                currency: 'SEK',

                            }
                        }).format(response.data.loan.loan_sum),
                        paymentPlan: response.data.loan.payment_plan,
                        years: response.data.loan.years
                    };

                    loanTemplate = Handlebars.compile(loanSource);
                    loanBox.html(loanTemplate(samblaData));
                } else {
                    $('#valuation-insurance-section').fadeOut();
                }

            })
            .catch(function (error) {
                valuationDiv.fadeOut();
                loanBox.fadeOut();
            });
    }

    $.valuation.getValuation();

<script type="text/javascript">
    $(document).ready(function () {
        var equipment = {};

        equipment.interval;
        equipment.jobId = null;
        equipment.isIntervalRunning = false;

        equipment.clearJobId = function () {
            equipment.jobId = null;
        }

        equipment.clearInterval = function () {
            clearInterval(equipment.interval);
        }

        equipment.sendRequest = function (postData) {
            $.ajax({
                type: 'POST',
                url: '/api/v1/oem/equipment/create',
                data: postData,
                beforeSend: function () {
                    $('#equipment-widget-status').html('<i class="fa fa-spinner fa-spin fa-1x"></i> Hämtar uppgifter från tillverkaren...');
                },
                success: function (response) {
                    equipment.jobId = response.job_id;
                    equipment.checkResponse();
                },
                error: function (request, status, error) {
                    if (request.status === 403) {
                        $('#equipment-widget-status').html('Det gick inte att hämta fler uppgifter, var vänlig vänta en stund.');
                    } else {
                        $('#equipment-widget-status').html('Det uppstod ett okänt problem, ladda om sidan och försök igen.');
                    }
                }
            });
        }

        equipment.checkResponse = function (postData) {
            equipment.isIntervalRunning = true;

            equipment.interval = setInterval(function () {
                if (equipment.jobId == null) { return; }

                $.ajax({
                    type: 'POST',
                    url: '/api/v1/oem/equipment/poll',
                    data: {
                        'job_id': equipment.jobId
                    },
                    beforeSend: function () {
                        $('#equipment-widget-status').html('<i class="fa fa-spinner fa-spin fa-1x"></i> Sammanställer uppgifter från tillverkaren...');
                    },
                    success: function (response) {
                        if (response.status && response.status == 2) {
                            return;
                        }
                        else if (response.status && response.status == 3) {
                            $('#equipment-widget-status').html('Klart! Du kan se rapporten genom att <a href="'+response.url+'" target="_blank" id="equipment-widget-view" class="font-weight-bold">klicka här</a>.');
                            equipment.clearJobId();
                            equipment.clearInterval();
                        }
                        else if (response.status && response.status == 4) {
                            $('#equipment-widget-status').html('Det uppstod ett okänt problem, ladda om sidan och försök igen.');
                            equipment.clearJobId();
                            equipment.clearInterval();
                        }
                        else if (response.status && (response.status == 5 || response.status == 6)) {
                            $('#equipment-widget-status').html('Detta fordon har ingen tillgänglig information. <a href="#" data-toggle="modal" data-target="#modal-equipment" id="equipment-widget-read-more" class="font-weight-bold">Läs mer</a>');
                            equipment.clearJobId();
                            equipment.clearInterval();
                        }
                    },
                    error: function (request, status, error) {
                        $('#equipment-widget-status').html('Det uppstod ett okänt problem, ladda om sidan och försök igen.');
                        equipment.clearJobId();
                        equipment.clearInterval();
                    }
                });
            }, 3000);
        }

        $(document).on('click', '#equipment-widget-read-more', function (e) {
            e.preventDefault();

            let modalId = $(this).attr('data-toggle');

            $(modalId).modal('show');
        });

        $(document).on('click', '#equipment-submit', function (e) {
            e.preventDefault();

            equipment.sendRequest({
                vehicle_id: 'o2WJLK'
            });
        });

        $(window).focus(function () {
            clearInterval(equipment.interval);

            if (equipment.jobId != null && equipment.isIntervalRunning == false)  {
                equipment.checkResponse();
            }
        }).blur(function () {
            clearInterval(equipment.interval);
            equipment.isIntervalRunning = false;
        });
    });
</script>
